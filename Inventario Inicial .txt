function buildForm() {
  var container = document.getElementById('productos-container');
  categorias.forEach(function(cat) {
    var card = document.createElement('div');
    card.className = 'card';
    card.innerHTML = '<div class="card-hdr"><div class="card-hdr-icon ' + cat.css + '">' + cat.icon + '</div><h2>' + cat.nombre + '</h2><span class="badge" id="badge-' + cat.id + '">0</span></div>';
    var prods = document.createElement('div');
    cat.productos.forEach(function(prod, idx) {
      var id = cat.id + '_' + idx;
      var row = document.createElement('div');
      row.className = 'prod-row';
      row.innerHTML =
        '<span class="prod-name">' + prod + '</span>' +
        '<div class="qty-wrap">' +
          '<button class="qty-btn minus" onclick="chg(\'' + id + '\',-1)">−</button>' +
          '<input class="qty-input" type="number" id="' + id + '" value="0" min="0" oninput="upd()">' +
          '<button class="qty-btn plus" onclick="chg(\'' + id + '\',1)">+</button>' +
        '</div>';
      prods.appendChild(row);
    });
    card.appendChild(prods);
    container.appendChild(card);
  });
}

function chg(id, d) {
  var el = document.getElementById(id);
  var v = parseInt(el.value) || 0;
  el.value = Math.max(0, v + d);
  upd();
}

function getV(id) { return parseInt(document.getElementById(id).value) || 0; }

function upd() {
  var grand = 0;
  categorias.forEach(function(cat) {
    var catTot = 0;
    cat.productos.forEach(function(_, idx) {
      var v = getV(cat.id + '_' + idx);
      catTot += v;
      var el = document.getElementById(cat.id + '_' + idx);
      el.className = 'qty-input' + (v > 0 ? ' has-val' : '');
    });
    grand += catTot;
    document.getElementById('badge-' + cat.id).textContent = catTot;
  });
  document.getElementById('grand-total').textContent = grand + ' unidades';
}

async function enviar() {
  if (!document.getElementById('sucursal').value) {
    document.getElementById('e-suc').classList.add('show');
    window.scrollTo(0, 0); return;
  }
  document.getElementById('e-suc').classList.remove('show');
  var btn = document.getElementById('btn-send');
  btn.textContent = '⏳ Enviando...'; btn.disabled = true;

  var productos = {};
  categorias.forEach(function(cat) {
    cat.productos.forEach(function(prod, idx) {
      productos[prod] = getV(cat.id + '_' + idx);
    });
  });

  var params = new URLSearchParams({
    tipo_formulario: "inventario_inicial",
    sucursal: document.getElementById('sucursal').value,
    fecha: document.getElementById('fecha').value,
    encargada: document.getElementById('encargada').value,
    productos_json: JSON.stringify(productos)
  });

  try {
    await fetch(SCRIPT_URL, { method: 'POST', mode: 'no-cors', body: params });
    document.querySelector('.warn-banner').style.display = 'none';
    document.getElementById('productos-container').style.display = 'none';
    document.querySelector('.card').style.display = 'none';
    document.querySelector('.navbtns').style.display = 'none';
    document.querySelector('.total-bar').style.display = 'none';
    document.getElementById('success').classList.add('show');
    window.scrollTo(0, 0);
  } catch(err) {
    btn.textContent = '📦 Registrar inventario inicial';
    btn.disabled = false;
    alert('Error al enviar. Revisá tu conexión e intentá de nuevo.');
  }
}

document.getElementById('fecha').valueAsDate = new Date();
